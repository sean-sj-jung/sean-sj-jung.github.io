import numpy as np

def kmeans(data, k, max_iters=100, tol=1e-4):
    X = np.array(data)
    
    # Initialize centroids by randomly picking k points from the data
    rng = np.random.default_rng(seed=42)
    indices = rng.choice(len(X), size=k, replace=False)
    centroids = X[indices]

    for _ in range(max_iters):
        # Assignment step: assign each point to the nearest centroid
        diffs = X[:, np.newaxis, :] - centroids[np.newaxis, :, :]  # (n, k, d)
        distances = np.linalg.norm(diffs, axis=2)                  # (n, k)
        labels = np.argmin(distances, axis=1)                       # (n,)

        # Update step: recompute centroids as the mean of assigned points, no empty cluster through else statement
        new_centroids = np.array([
            X[labels == i].mean(axis=0) if np.any(labels == i) else centroids[i]
            for i in range(k)
        ]) 

        # Convergence check
        if np.linalg.norm(new_centroids - centroids) < tol:
            break

        centroids = new_centroids

    return labels, centroids
