import numpy as np


class KMeans:
    """K-Means clustering implemented from scratch."""

    def __init__(self, k=3, max_iters=100, tol=1e-4, random_state=None):
        self.k = k
        self.max_iters = max_iters
        self.tol = tol
        self.random_state = random_state
        self.centroids = None
        self.labels = None

    def _init_centroids(self, X):
        """K-Means++ initialization for better convergence."""
        rng = np.random.default_rng(self.random_state)
        n_samples = X.shape[0]

        # Pick first centroid randomly
        centroids = [X[rng.integers(n_samples)]]

        for _ in range(1, self.k):
            # Squared distances to nearest existing centroid
            dists = np.array([
                np.min(np.sum((X - c) ** 2, axis=1)) for c in centroids
            ]).T  # shape: (n_samples, num_centroids_so_far) -> min across cols
            # Actually we want min dist per point to ANY centroid
            dists = np.min(
                np.array([np.sum((X - c) ** 2, axis=1) for c in centroids]),
                axis=0,
            )
            # Sample proportional to distance squared
            probs = dists / dists.sum()
            idx = rng.choice(n_samples, p=probs)
            centroids.append(X[idx])

        return np.array(centroids)

    def _assign_clusters(self, X):
        """Assign each point to the nearest centroid."""
        # Vectorized distance: (n_samples, 1, n_features) - (1, k, n_features)
        diffs = X[:, np.newaxis, :] - self.centroids[np.newaxis, :, :]
        dists = np.sum(diffs ** 2, axis=2)  # (n_samples, k)
        return np.argmin(dists, axis=1)

    def _update_centroids(self, X, labels):
        """Recompute centroids as mean of assigned points."""
        new_centroids = np.zeros_like(self.centroids)
        for i in range(self.k):
            members = X[labels == i]
            if len(members) > 0:
                new_centroids[i] = members.mean(axis=0)
            else:
                # Handle empty cluster: reinitialize randomly
                new_centroids[i] = self.centroids[i]
        return new_centroids

    def fit(self, X):
        """Run K-Means until convergence or max iterations."""
        self.centroids = self._init_centroids(X)

        for _ in range(self.max_iters):
            self.labels = self._assign_clusters(X)
            new_centroids = self._update_centroids(X, self.labels)

            # Check convergence: max centroid shift
            shift = np.max(np.sum((new_centroids - self.centroids) ** 2, axis=1))
            self.centroids = new_centroids
            if shift < self.tol:
                break

        self.inertia_ = self._compute_inertia(X)
        return self

    def predict(self, X):
        """Assign new points to nearest centroid."""
        return self._assign_clusters(X)

    def _compute_inertia(self, X):
        """Sum of squared distances to assigned centroid (WCSS)."""
        dists = np.sum((X - self.centroids[self.labels]) ** 2)
        return dists


# ---------- Demo ----------
if __name__ == "__main__":
    np.random.seed(42)
    # Generate 3 blobs
    blob1 = np.random.randn(100, 2) + np.array([0, 0])
    blob2 = np.random.randn(100, 2) + np.array([5, 5])
    blob3 = np.random.randn(100, 2) + np.array([10, 0])
    X = np.vstack([blob1, blob2, blob3])

    km = KMeans(k=3, random_state=42)
    km.fit(X)

    print(f"Centroids:\n{km.centroids}")
    print(f"Inertia (WCSS): {km.inertia_:.2f}")
    print(f"Label counts: {np.bincount(km.labels)}")```
